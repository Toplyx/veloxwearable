// Uložíme produkty do košíku do localStorage (můžeme použít i jiný storage podle potřeby) let cart = JSON.parse(localStorage.getItem('cart')) || []; // Funkce pro přidání produktu do košíku function addToCart(productId) { // Zde můžete přidat produkt dle jeho ID a dalších vlastností const product = { id: productId, name: `Produkt ${productId}`, price: `${productId * 100} CZK` }; cart.push(product); localStorage.setItem('cart', JSON.stringify(cart)); // Uložíme košík do localStorage } // Funkce pro zobrazení produktů v košíku function displayCart() { const cartItems = document.getElementById('cart-items'); cartItems.innerHTML = ''; // Vymažeme obsah, pokud už nějaké produkty jsou if (cart.length === 0) { cartItems.innerHTML = '
  • Váš košík je prázdný.
    • '; } else { cart.forEach(item => { const li = document.createElement('li'); li.textContent = `${item.name} - ${item.price}`; cartItems.appendChild(li); }); } document.getElementById('cart-modal').style.display = 'flex'; // Zobrazíme košík } // Funkce pro zavření košíku function closeCart() { document.getElementById('cart-modal').style.display = 'none'; // Skryjeme košík } // Přidání event listenerů pro tlačítka "Přidat do košíku" const addToCartButtons = document.querySelectorAll('.add-to-cart'); addToCartButtons.forEach(button => { button.addEventListener('click', function() { const productId = this.parentElement.getAttribute('data-id'); // Získáme ID produktu addToCart(productId); }); }); // Přidání event listeneru pro tlačítko "Košík" document.getElementById('cart-btn').addEventListener('click', function(event) { event.preventDefault(); // Zabráníme defaultnímu chování odkazu displayCart(); // Zobrazíme košík }); // Přidání event listeneru pro zavření košíku document.getElementById('close-cart-btn').addEventListener('click', closeCart);
